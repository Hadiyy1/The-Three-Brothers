		name = sweden_flavor.1.a

		}

		name = sweden_flavor.1.b

			base = 1
		}
		remove_modifier = sweden_land_based_taxation_policies
		every_interest_group = {
			limit = {
				is_marginal = no
				NOT = { has_ideology = ideology_particularist }
				NOT = { has_ideology = ideology_jingoist }
